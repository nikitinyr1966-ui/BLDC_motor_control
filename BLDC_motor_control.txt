void UART_sendFloat(float value)
{
    uint16_t temp[2];
    uint16_t sendBuffer[4];
    memcpy(temp, &value, 2);
    sendBuffer[3] = temp[1] >> 8;
    sendBuffer[2] = temp[1] & 0x00FF;
    sendBuffer[1] = temp[0] >> 8;
    sendBuffer[0] = temp[0] & 0x00FF;
    SCI_writeCharArray(SCIA_BASE, sendBuffer, 4);
}

void initCPUTimer(void)
{
    CPUTimer_setPeriod(CPUTIMER0_BASE, 0xFFFFFFFF);
    CPUTimer_setPreScaler(CPUTIMER0_BASE, 0);
    CPUTimer_stopTimer(CPUTIMER0_BASE);
    CPUTimer_reloadTimerCounter(CPUTIMER0_BASE);
}

void mySPI0_init(){
    SPI_disableModule(mySPI0_BASE);
    SPI_setConfig(mySPI0_BASE, DEVICE_LSPCLK_FREQ, SPI_PROT_POL0PHA0,
                  SPI_MODE_MASTER, 1000000, 16);
    SPI_disableFIFO(mySPI0_BASE);
    SPI_disableLoopback(mySPI0_BASE);
    SPI_setEmulationMode(mySPI0_BASE, SPI_EMULATION_STOP_AFTER_TRANSMIT);
    SPI_enableModule(mySPI0_BASE);
}

void configCPUTimer(uint32_t cpuTimer, float freq, float period)
{
    uint32_t temp;
    temp = (uint32_t)(freq / 1000000 * period);
    CPUTimer_setPeriod(cpuTimer, temp);
    CPUTimer_setPreScaler(cpuTimer, 0);
    CPUTimer_stopTimer(cpuTimer);
    CPUTimer_reloadTimerCounter(cpuTimer);
    CPUTimer_setEmulationMode(cpuTimer,
                              CPUTIMER_EMULATIONMODE_STOPAFTERNEXTDECREMENT);
    CPUTimer_enableInterrupt(cpuTimer);
}

{
    ADC_setPrescaler(ADCA_BASE, ADC_CLK_DIV_1_0);
    ADC_setPrescaler(ADCB_BASE, ADC_CLK_DIV_1_0);
    ADC_setPrescaler(ADCC_BASE, ADC_CLK_DIV_1_0);

    ADC_setMode(ADCA_BASE, EX_ADC_RESOLUTION, EX_ADC_SIGNAL_MODE);
    ADC_setMode(ADCB_BASE, EX_ADC_RESOLUTION, EX_ADC_SIGNAL_MODE);
    ADC_setMode(ADCC_BASE, EX_ADC_RESOLUTION, EX_ADC_SIGNAL_MODE);

    ADC_enableConverter(ADCA_BASE);
    ADC_enableConverter(ADCB_BASE);
    ADC_enableConverter(ADCC_BASE);
    DEVICE_DELAY_US(1000);
}

    if (timer_engine_counter == timer_engine)
    {
        EPWM1_CMP = EPWM1_TIMER_TBPRD*((KM*((sinf(angle_counter*PI/180.0f)+1.0f)/2.0f)));
        EPWM2_CMP = EPWM2_TIMER_TBPRD*((KM*((sinf(angle_counter*PI/180.0f+(2*PI/3))+1.0f)/2.0f)));
        EPWM3_CMP = EPWM3_TIMER_TBPRD*((KM*((sinf(angle_counter*PI/180.0f+(4*PI/3))+1.0f)/2.0f)));
        angle_counter++;
        if (angle_counter == 360) angle_counter = 0;
        timer_engine_counter = 0;
    }

    timer_f_counter++;
    if (timer_f_counter == timer_f)
    {
        if (timer_engine > f_min)
        {
            timer_engine -= f_delta;
            volt = k / timer_engine;
            timer_engine_counter = 0;
        }
        timer_f_counter = 0;
    }

    timer_SPI_counter++;
    if (timer_SPI_counter == timer_SPI)
    {
        SPI_writeDataNonBlocking(mySPI0_BASE, 0x0);

        rData = SPI_readDataNonBlocking(mySPI0_BASE);
        rData =  rData & ~0xC000;
        enc_pos = rData / 16382.0f * 360.0f;
        if (fabsf(enc_pos_prev - enc_pos) > 270)
        {
            if (enc_pos_prev > enc_pos)
            {
                deltaPos = enc_pos - enc_pos_prev + 360.0f;
            }
            else
            {
                deltaPos = enc_pos - enc_pos_prev - 360.0f;
            }
        }
        else
        {
            deltaPos = enc_pos - enc_pos_prev;
        }
        engine_speed = deltaPos / 0.001f;
        enc_pos_prev = enc_pos;

        if(index_buf < 2000)
        {
            adcResult_A_v = ADC_readResult(ADCARESULT_BASE, ADC_SOC_NUMBER1);
            adcResult_B_v = ADC_readResult(ADCCRESULT_BASE, ADC_SOC_NUMBER0);
            adcResult_C_v = ADC_readResult(ADCBRESULT_BASE, ADC_SOC_NUMBER2);

            buf_Va[index_buf] = (float)(adcResult_A_v / 4096.0f) * 3.3f;
            buf_Vb[index_buf] = (float)(adcResult_B_v / 4096.0f) * 3.3f;
            buf_Vc[index_buf] = (float)(adcResult_C_v / 4096.0f) * 3.3f;
            buf_speed[index_buf] = engine_speed;
            buf_pos[index_buf] = enc_pos;
            index_buf++;
        }

        /*
